enum class DayOfTheWeek {
	Sunday,
	Monday,
	Tuesday,
	Wednesday,
	Thursday,
	Friday,
	Saturday
}

main function -> 
	for (day in DayOfTheWeek.values()) {
 	 println("Day ${day.ordinal}: ${day.name}")
		}

Day 2: Tuesday

Day 4: Thursday
Day 5: Friday
Day 6: Saturday
